git add [檔案或資料夾]	將指定檔案或資料夾加入版本控制

git add -A 加入全部

git commit -m	"說明文字" 提交異動並設定說明文字

git log		查看先前的commit記錄

git log --oneline		只顯示commit訊息和hash value

git push	將本地端的Repository的commit發佈到遠端

git push origin [branch_name]	發佈到指定分支

git branch	查看分支

git branch [branch_name]	建立分支

git checkout [branch_name]	取出指定分支

git checkout [HASH_VALUE]	切換到指定的commit
